# Setup Git per prompt (solo se necessario)
setup_git_prompt() {
    autoload -Uz vcs_info
    precmd_vcs_info() { vcs_info }
    precmd_functions+=( precmd_vcs_info )
    zstyle ':vcs_info:*' enable git
    zstyle ':vcs_info:*' formats ' (%b%u%c)'
    zstyle ':vcs_info:*' actionformats ' (%b|%a%u%c)'
    zstyle ':vcs_info:git:*' check-for-changes true
    zstyle ':vcs_info:git:*' unstagedstr '*'
    zstyle ':vcs_info:git:*' stagedstr '+'
}
# Prompt Base (no git, no progetto)
set_base_prompt() {
    PROMPT='%F{blue}[%D{%Y-%m-%d %H:%M}]%f %F{green}%n@$(scutil --get ComputerName 2>/dev/null || hostname -s)%f %F{yellow}%~%f
%# '
    # Rimuovi git info
    precmd_functions=()
}

# ===================================================================
# FUNZIONE PER CAMBIARE PROGETTO
# ===================================================================
switch_project() {
    local choice=$1
    case $choice in
        1|rcc)
            echo "🎮 Caricamento ambiente Revolution Chess Clock..."
            export CURRENT_PROJECT="rcc"
            # Carica environment RCC
            RCC_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/ChessClock/GitHub/RCCS2025/src/shell_per_il_deploy/rcc_2025_env.zsh"
            if [[ -f "$RCC_ENV_FILE" ]]; then
                source "$RCC_ENV_FILE"
                set_rcc_prompt
                echo "✅ Ambiente RCC caricato!"
                echo "📝 Digita 'rcchelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment RCC non trovato: $RCC_ENV_FILE"
                set_base_prompt
            fi
            ;;
        2|gdm)
            echo "♟️  Caricamento ambiente Game Director Manager..."
            export CURRENT_PROJECT="gdm"
            # Carica environment GDM
            GDM_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/ChessClock/GitHub/RCGDM2025/src/shell_per_il_deploy/gdm_2025_env.zsh"
            if [[ -f "$GDM_ENV_FILE" ]]; then
                source "$GDM_ENV_FILE"
                set_gdm_prompt
                echo "✅ Ambiente GDM caricato!"
                echo "📝 Digita 'gdmhelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment GDM non trovato: $GDM_ENV_FILE"
                set_base_prompt
            fi
            ;;
        3|www)
            echo "🌐 Caricamento ambiente ivanluminaria.com..."
            export CURRENT_PROJECT="www"
            # Variabili progetto WWW
            export WWW_PROJECT_DIR="/Users/ivanluminaria/Development/APPLICAZIONI/My-Web-Site/ivanluminaria.com"
            export WWW_CONTENT_DIR="$WWW_PROJECT_DIR/content"
            export WWW_CONFIG_DIR="$WWW_PROJECT_DIR/config/_default"
            export WWW_DOCS_DIR="$WWW_PROJECT_DIR/DOCS"
            export HUGO_BASE_URL="https://ilum75IDB.github.io/ivanluminaria.com/"
            export WWW_PROD_URL="https://www.ivanluminaria.com/"
            export WWW_REPO="ilum75IDB/ivanluminaria.com"
            if [[ -d "$WWW_PROJECT_DIR" ]]; then
                cd "$WWW_PROJECT_DIR"
                set_www_prompt
                # Alias specifici Hugo/sito
                alias hserve='hugo server -D --navigateToChanged'
                alias hbuild='hugo --minify'
                alias hclean='rm -rf public/ resources/_gen/ .hugo_build.lock'
                alias hnew='hugo new'
                # Navigazione rapida
                alias cdwww='cd $WWW_PROJECT_DIR'
                alias cdcontent='cd $WWW_CONTENT_DIR'
                alias cdposts='cd $WWW_CONTENT_DIR/posts'
                alias cdresumes='cd $WWW_CONTENT_DIR/resumes'
                alias cdconfig='cd $WWW_CONFIG_DIR'
                alias cddocs='cd $WWW_DOCS_DIR'
                alias cdcss='cd $WWW_PROJECT_DIR/assets/css'
                alias cdlayouts='cd $WWW_PROJECT_DIR/layouts'
                # Git shortcuts per il sito
                alias gstatus='git status'
                alias gdiff='git diff'
                alias glog='git log --oneline -20'
                alias gpush='git push origin main'
                alias gpull='git pull origin main'
                # Deploy & preview
                alias deploy='git add -A && git commit -m "update site" && git push origin main'
                alias preview='open "https://ilum75IDB.github.io/ivanluminaria.com/"'
                alias ghactions='open "https://github.com/ilum75IDB/ivanluminaria.com/actions"'
                # Contenuti multilingua - crea set di file per nuova pagina
                www_new_page() {
                    local name=$1
                    local dir=${2:-"content"}
                    if [[ -z "$name" ]]; then
                        echo "❌ Uso: www_new_page <nome> [directory]"
                        echo "   Esempio: www_new_page about content"
                        return 1
                    fi
                    for lang in it en es ro; do
                        local filepath="$WWW_PROJECT_DIR/$dir/${name}.${lang}.md"
                        if [[ ! -f "$filepath" ]]; then
                            echo "---" > "$filepath"
                            echo "title: \"\"" >> "$filepath"
                            echo "date: $(date +%Y-%m-%dT%H:%M:%S%z)" >> "$filepath"
                            echo "draft: true" >> "$filepath"
                            echo "---" >> "$filepath"
                            echo "✅ Creato: $filepath"
                        else
                            echo "⚠️  Esiste già: $filepath"
                        fi
                    done
                }
                # Contenuti multilingua - crea set di file per nuovo post
                www_new_post() {
                    local name=$1
                    if [[ -z "$name" ]]; then
                        echo "❌ Uso: www_new_post <slug-del-post>"
                        echo "   Esempio: www_new_post oracle-vs-postgres"
                        return 1
                    fi
                    www_new_page "$name" "content/posts"
                }
                # Submodule update per il tema Congo
                alias themeupdate='cd $WWW_PROJECT_DIR && git submodule update --init --recursive'
                echo "✅ Ambiente WWW caricato!"
                echo "📝 Digita 'wwwhelp' per vedere i comandi disponibili"
            else
                echo "❌ Directory progetto non trovata: $WWW_PROJECT_DIR"
                set_base_prompt
            fi
            ;;
        4|none)
            echo "🔧 Ambiente base (nessun progetto)"
            export CURRENT_PROJECT="none"
            set_base_prompt
            echo "✅ Ambiente base caricato!"
            ;;
        *)
            echo "❌ Scelta non valida"
            ;;
    esac
}

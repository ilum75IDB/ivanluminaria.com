# Setup Git per prompt (solo se necessario)
setup_git_prompt() {
    autoload -Uz vcs_info
    precmd_vcs_info() { vcs_info }
    precmd_functions+=( precmd_vcs_info )
    zstyle ':vcs_info:*' enable git
    zstyle ':vcs_info:*' formats ' (%b%u%c)'
    zstyle ':vcs_info:*' actionformats ' (%b|%a%u%c)'
    zstyle ':vcs_info:git:*' check-for-changes true
    zstyle ':vcs_info:git:*' unstagedstr '*'
    zstyle ':vcs_info:git:*' stagedstr '+'
}
# Prompt Base (no git, no progetto)
set_base_prompt() {
    PROMPT='%F{blue}[%D{%Y-%m-%d %H:%M}]%f %F{green}%n@$(scutil --get ComputerName 2>/dev/null || hostname -s)%f %F{yellow}%~%f
%# '
    # Rimuovi git info
    precmd_functions=()
}

switch_project() {
    local choice=$1
    case $choice in
        1|rcc)
            echo "🎮 Caricamento ambiente Revolution Chess Clock..."
            export CURRENT_PROJECT="rcc"
            # Carica environment RCC
            RCC_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/ChessClock/GitHub/RCCS2025/src/shell_per_il_deploy/rcc_2025_env.zsh"
            if [[ -f "$RCC_ENV_FILE" ]]; then
                source "$RCC_ENV_FILE"
                set_rcc_prompt
                echo "✅ Ambiente RCC caricato!"
                echo "📝 Digita 'rcchelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment RCC non trovato: $RCC_ENV_FILE"
                set_base_prompt
            fi
            ;;
        2|gdm)
            echo "♟️  Caricamento ambiente Game Director Manager..."
            export CURRENT_PROJECT="gdm"
            # Carica environment GDM
            GDM_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/ChessClock/GitHub/RCGDM2025/src/shell_per_il_deploy/gdm_2025_env.zsh"
            if [[ -f "$GDM_ENV_FILE" ]]; then
                source "$GDM_ENV_FILE"
                set_gdm_prompt
                echo "✅ Ambiente GDM caricato!"
                echo "📝 Digita 'gdmhelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment GDM non trovato: $GDM_ENV_FILE"
                set_base_prompt
            fi
            ;;
        3|www)
            echo "🌐 Caricamento ambiente ivanluminaria.com..."
            export CURRENT_PROJECT="www"
            # Carica environment WWW (pattern modulare come RCC/GDM/WEBO)
            WWW_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/My-Web-Site/ivanluminaria.com/scripts/shell/www_env.zsh"
            if [[ -f "$WWW_ENV_FILE" ]]; then
                source "$WWW_ENV_FILE"
                cd "$WWW_PROJECT_DIR"
                set_www_prompt
                echo "✅ Ambiente WWW caricato!"
                echo "📝 Digita 'wwwhelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment WWW non trovato: $WWW_ENV_FILE"
                set_base_prompt
            fi
            ;;
        4|gma)
            echo "📧 Caricamento ambiente Gmail Assistant AI Agent..."
            export CURRENT_PROJECT="gma"
            # Variabili progetto GMA
            export GMA_PROJECT_DIR="/Users/ivanluminaria/Development/APPLICAZIONI/Gmail-Asistant-AI-Agent"
            export GMA_VENV_DIR="$GMA_PROJECT_DIR/.venv"
            export GMA_REPO="ilum75IDB/gmail-asistant-ai-agent"
            export GMA_PORT="8000"
            if [[ -d "$GMA_PROJECT_DIR" ]]; then
                cd "$GMA_PROJECT_DIR"
                set_gma_prompt
                # Attiva il venv Python se esiste
                if [[ -f "$GMA_VENV_DIR/bin/activate" ]]; then
                    source "$GMA_VENV_DIR/bin/activate"
                fi
                # Navigazione rapida
                alias cdgma='cd $GMA_PROJECT_DIR'
                alias cdgmaapp='cd $GMA_PROJECT_DIR/app'
                alias cdgmadata='cd $GMA_PROJECT_DIR/data'
                alias cdgmadash='cd $GMA_PROJECT_DIR/dashboard'
                alias cdgmamockups='cd $GMA_PROJECT_DIR/mockups'
                alias cdgmascripts='cd $GMA_PROJECT_DIR/scripts'
                # Server FastAPI
                alias gmaserver='uvicorn app.main:app --reload --port $GMA_PORT'
                alias gmaopen='open http://127.0.0.1:$GMA_PORT/docs'
                alias gmahealth='curl -s http://127.0.0.1:$GMA_PORT/api/health | python -m json.tool'
                alias gmastatus='curl -s http://127.0.0.1:$GMA_PORT/api/diagnostics | python -m json.tool'
                # Database e seed
                alias gmaseed='python -m scripts.seed_db'
                alias gmaseedreset='python -m scripts.seed_db --reset'
                alias gmadb='sqlite3 $GMA_PROJECT_DIR/data/state.db'
                # Git shortcuts
                alias gstatus='git status'
                alias gdiff='git diff'
                alias glog='git log --oneline -20'
                alias gpush='git push'
                alias gpull='git pull'
                echo "✅ Ambiente GMA caricato!"
                if [[ -n "$VIRTUAL_ENV" ]]; then
                    echo "🐍 venv Python attivo: $VIRTUAL_ENV"
                else
                    echo "⚠️  venv non trovato. Crealo con: python3 -m venv .venv && source .venv/bin/activate && pip install -r requirements.txt"
                fi
                echo "📝 Digita 'gmahelp' per vedere i comandi disponibili"
            else
                echo "❌ Directory progetto non trovata: $GMA_PROJECT_DIR"
                set_base_prompt
            fi
            ;;
        5|webo)
            echo "🎛  Caricamento ambiente Web Orchestrator..."
            export CURRENT_PROJECT="webo"
            # Carica environment WEBO (pattern modulare come RCC/GDM)
            WEBO_ENV_FILE="/Users/ivanluminaria/Development/APPLICAZIONI/My-Web-Site/web-orchestrator/scripts/shell/webo_env.zsh"
            if [[ -f "$WEBO_ENV_FILE" ]]; then
                source "$WEBO_ENV_FILE"
                set_webo_prompt
                echo "✅ Ambiente WEBO caricato!"
                echo "📝 Digita 'webohelp' per vedere i comandi disponibili"
            else
                echo "❌ File environment WEBO non trovato: $WEBO_ENV_FILE"
                set_base_prompt
            fi
            ;;
        6|none)
            echo "🔧 Ambiente base (nessun progetto)"
            export CURRENT_PROJECT="none"
            set_base_prompt
            echo "✅ Ambiente base caricato!"
            ;;
        *)
            echo "❌ Scelta non valida"
            ;;
    esac
}

    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo "🚀 SELEZIONA AMBIENTE DI LAVORO"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo ""
    echo "  1) 🎮 Revolution Chess Clock   (RCC)"
    echo "  2) ♟️  Game Director Manager   (GDM)"
    echo "  3) 🌐 www.ivanluminaria.com    (WWW)"
    echo "  4) 📧 Gmail Assistant AI Agent (GMA)"
    echo "  5) 🎛  Web Orchestrator        (WEBO)"
    echo "  6) 🔧 Ambiente Base            (BASE)"
    echo ""
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo -n "Scegli (1-6) o premi INVIO per ambiente base: "
    read -t 10 choice  # Timeout di 10 secondi
    if [[ -z "$choice" ]]; then
        choice=6  # Default: ambiente base
    fi
    echo ""
    switch_project $choice
    echo ""
}

# ===================================================================
help() {
    echo ""
    echo "🎯 COMANDI DISPONIBILI"
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
    echo ""
    echo "📂 CAMBIO PROGETTO:"
    echo "  workrcc         # Passa a Revolution Chess Clock"
    echo "  workgdm         # Passa a Game Director Manager"
    echo "  workwww         # Passa a www.ivanluminaria.com"
    echo "  workgma         # Passa a Gmail Assistant AI Agent"
    echo "  workwebo        # Passa a Web Orchestrator"
    echo "  workbase        # Passa ad ambiente base"
    echo "  pstatus         # Mostra progetto attivo"
    echo ""
    echo "❓ HELP SPECIFICI:"
    case $CURRENT_PROJECT in
        rcc)
            echo "  rcchelp         # Help comandi RCC"
            ;;
        gdm)
            echo "  gdmhelp         # Help comandi GDM"
            ;;
        www)
            echo "  wwwhelp         # Help comandi WWW"
            ;;
        gma)
            echo "  gmahelp         # Help comandi GMA"
            ;;
        webo)
            echo "  webohelp        # Help comandi WEBO"
            ;;
        *)
            echo "  (carica un progetto per vedere i comandi specifici)"
            ;;
    esac
    echo ""
    echo "🎨 UTILITY:"
    echo "  lscolors        # Mostra legenda colori ls"
    echo "  show_ls_themes  # Mostra temi colori disponibili"
    echo "  set_ls_theme N  # Cambia tema colori ls"
    echo ""
    echo "📁 ALIAS LS SPECIALI:"
    echo "  lsd             # Solo directory"
    echo "  lsf             # Solo file normali"
    echo "  lsx             # Solo file eseguibili"
    echo "  lsl             # Solo link simbolici"
    echo ""
    echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
}
# ===================================================================
# INIZIALIZZAZIONE
# ===================================================================
# Se è una shell interattiva, mostra il menu
if [[ -o interactive ]]; then
    # Se non è già stato scelto un progetto (es. nuovo terminale)
    if [[ "$CURRENT_PROJECT" == "none" ]] || [[ -z "$CURRENT_PROJECT" ]]; then
        show_project_menu
    else
        # Ricarica l'ambiente del progetto corrente
        switch_project $CURRENT_PROJECT
    fi
fi
# Se tutto va male, usa il prompt base
if [[ -z "$PROMPT" ]]; then
    set_base_prompt
fi

# npm global packages
export PATH=~/.npm-global/bin:$PATH
